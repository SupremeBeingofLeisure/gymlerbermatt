		\item $\emptyset\in\mathcal{P}(\emptyset)$
		\item $\N\cap\Q=\N_0$

		\item $\N_0\setminus\N=\{0\}$

		\item $\mathcal{P}(\{1,2\})\cap\N=\emptyset$

	\end{multicols}
	\end{enumerate}
